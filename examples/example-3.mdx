## Overview

This example demonstrates how to integrate our product with third-party services and systems. We'll show you how to connect with popular platforms and extend functionality through integrations.

## Integration with Authentication Providers

### OAuth Integration

```typescript
// Configure OAuth integration
const oauthConfig = {
  provider: 'google',
  clientId: process.env.GOOGLE_CLIENT_ID,
  clientSecret: process.env.GOOGLE_CLIENT_SECRET,
  redirectUri: 'https://your-app.com/auth/callback',
  scopes: ['profile', 'email']
};

// Initialize the client with OAuth
const client = new Client({
  apiKey: process.env.API_KEY,
  auth: {
    type: 'oauth',
    config: oauthConfig
  }
});

// Get the authorization URL
const authUrl = client.getAuthorizationUrl();
console.log('Redirect user to:', authUrl);

// Handle the callback (in your redirect handler)
async function handleOAuthCallback(code) {
  try {
    const tokens = await client.exchangeCodeForTokens(code);
    
    // Store tokens securely
    saveTokensToSecureStorage(tokens);
    
    // Client is now authenticated
    const userInfo = await client.getUserInfo();
    console.log('Authenticated user:', userInfo);
    
    return userInfo;
  } catch (error) {
    console.error('OAuth error:', error);
    throw error;
  }
}
```

## Integration with Data Storage Services

### AWS S3 Integration

```typescript
// Configure S3 integration
const s3Config = {
  accessKeyId: process.env.AWS_ACCESS_KEY_ID,
  secretAccessKey: process.env.AWS_SECRET_ACCESS_KEY,
  region: 'us-west-2',
  bucket: 'your-bucket-name'
};

// Initialize the client with storage integration
const client = new Client({
  apiKey: process.env.API_KEY,
  storage: {
    type: 's3',
    config: s3Config
  }
});

// Process and store results directly to S3
const result = await client.processAndStore({
  data: inputData,
  outputPath: 'results/job-123.json',
  options: {
    format: 'json',
    compression: 'gzip'
  }
});

console.log('Processing complete, stored at:', result.storageUri);

// Retrieve and process data from S3
const retrievedData = await client.retrieveAndProcess({
  storagePath: 's3://your-bucket-name/path/to/input.json',
  options: {
    transform: true
  }
});

console.log('Retrieved and processed data:', retrievedData);
```

## Integration with Messaging Services

### Webhook Integration

```typescript
// Register webhooks for event notifications
await client.registerWebhook({
  url: 'https://your-app.com/webhooks/events',
  events: ['job.completed', 'job.failed', 'user.action'],
  secret: process.env.WEBHOOK_SECRET
});

// In your webhook handler (Express.js example)
app.post('/webhooks/events', (req, res) => {
  // Verify webhook signature
  const signature = req.headers['x-webhook-signature'];
  const isValid = client.verifyWebhookSignature(
    signature,
    req.body,
    process.env.WEBHOOK_SECRET
  );
  
  if (!isValid) {
    return res.status(401).send('Invalid signature');
  }
  
  // Process the webhook event
  const event = req.body;
  console.log('Received webhook event:', event.type);
  
  switch (event.type) {
    case 'job.completed':
      handleJobCompleted(event.data);
      break;
    case 'job.failed':
      handleJobFailed(event.data);
      break;
    case 'user.action':
      handleUserAction(event.data);
      break;
    default:
      console.log('Unhandled event type:', event.type);
  }
  
  res.status(200).send('Event received');
});
```

## Integration with Analytics Services

// Configure analytics integration
const analyticsConfig = {

## Complete Integration Example

Here's a more complete example showing multiple integrations working together:

```typescript
// Main integration setup
async function setupIntegratedSystem() {
  // Initialize with multiple integrations
  const client = new Client({
    apiKey: process.env.API_KEY,
    
    // Authentication
    auth: {
      type: 'oauth',
      config: {
        provider: 'google',
        clientId: process.env.GOOGLE_CLIENT_ID,
        clientSecret: process.env.GOOGLE_CLIENT_SECRET,
        redirectUri: 'https://your-app.com/auth/callback'
      }
    },
    
    // Storage
    storage: {
      type: 's3',
      config: {
        accessKeyId: process.env.AWS_ACCESS_KEY_ID,
        secretAccessKey: process.env.AWS_SECRET_ACCESS_KEY,
        region: 'us-west-2',
        bucket: 'your-bucket-name'
      }
    },
    
    // Analytics
    analytics: {
      provider: 'segment',
      writeKey: process.env.SEGMENT_WRITE_KEY
    },
    
    // Notifications
    notifications: {
      email: {
        enabled: true,
        provider: 'sendgrid',
        apiKey: process.env.SENDGRID_API_KEY
      },
      slack: {
        enabled: true,
        webhookUrl: process.env.SLACK_WEBHOOK_URL
      }
    }
  });
  
  // Register webhooks
  await client.registerWebhook({
    url: 'https://your-app.com/webhooks/events',
    events: ['job.completed', 'job.failed'],
    secret: process.env.WEBHOOK_SECRET
  });
  
  return client;
}

// Usage example
async function runIntegratedWorkflow(userId, inputData) {
  const client = await setupIntegratedSystem();
  
  // Authenticate user
  await client.authenticateUser(userId);
  
  // Process data and store in S3
  const result = await client.processAndStore({
    data: inputData,
    outputPath: `users/${userId}/results/job-${Date.now()}.json`,
    options: {
      format: 'json',
      notify: true  // Send notification when complete
    }
  });
  
  // Track the event
  await client.trackEvent({
    event: 'Workflow Completed',
    properties: {
      userId,
      jobId: result.jobId,
      duration: result.duration,
      outputSize: result.size
    }
  });
  
  return result;
}
```
